253  private static void GenerateUriParameters(HelpPageApiModel apiModel, ModelDescriptionGenerator modelGenerator)
254  {
255  ApiDescription apiDescription = apiModel.ApiDescription;
256  foreach (ApiParameterDescription apiParameter in apiDescription.ParameterDescriptions)
257  {
258  if (apiParameter.Source == ApiParameterSource.FromUri)
259  {
260  HttpParameterDescriptor parameterDescriptor = apiParameter.ParameterDescriptor;
261  Type parameterType = null;
262  ModelDescription typeDescription = null;
263  ComplexTypeModelDescription complexTypeDescription = null;
264  if (parameterDescriptor != null)
265  {
266  parameterType = parameterDescriptor.ParameterType;
267  typeDescription = modelGenerator.GetOrCreateModelDescription(parameterType);
268  complexTypeDescription = typeDescription as ComplexTypeModelDescription;
269  }
270 
271  if (complexTypeDescription != null)
272  {
273  foreach (ParameterDescription uriParameter in complexTypeDescription.Properties)
274  {
275  apiModel.UriParameters.Add(uriParameter);
276  }
277  }
278  else if (parameterDescriptor != null)
279  {
280  ParameterDescription uriParameter =
281  AddParameterDescription(apiModel, apiParameter, typeDescription);
282 
283  if (!parameterDescriptor.IsOptional)
284  {
285  uriParameter.Annotations.Add(new ParameterAnnotation() { Documentation = "Required" });
286  }
287 
288  object defaultValue = parameterDescriptor.DefaultValue;
289  if (defaultValue != null)
290  {
291  uriParameter.Annotations.Add(new ParameterAnnotation() { Documentation = "Default value is " + Convert.ToString(defaultValue, CultureInfo.InvariantCulture) });
292  }
293  }
294  else
295  {
296  Debug.Assert(parameterDescriptor == null);
297 
298  // If parameterDescriptor is null, this is an undeclared route parameter which only occurs
299  // when source is FromUri. Ignored in request model and among resource parameters but listed
300  // as a simple string here.
301  ModelDescription modelDescription = modelGenerator.GetOrCreateModelDescription(typeof(string));
302  AddParameterDescription(apiModel, apiParameter, modelDescription);
303  }
304  }
305  }
